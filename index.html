Verbesserungen ES6

Object literal

{
  calculateSomething(a, b) {
    return a * b;
  }
} // Method properties

var a = 2, b = 2;
{ a, b } === { a: a, b: b } // Property shorthand
{ [ 'foo' + a ]: b } // Computed property names

Class literal

class Shape {
  constructor(height, width) {
    this.height = height;
    this.width = width;
  }
}

Class literal II / Methods

class Shape {
  constructor(height, width) { … }
  calculateArea() {
    return this.height * this.width;
  }
  get area() {
    return this.calculateArea();
  }
}

Class literal III / Static Methods

class Shape {
  static distance(a, b) {
    return Math.sqrt(
      (a.x - b.x)*(a.x - b.x) +
      (a.y - b.y)*(a.y - b.y)
    );
  }
}

Class literal IV / Inheritance

class Square extends Shape {
  constructor(a) {
    super(a, a);
  }
  myMethod() {
    return super.myMethod();
  }
}
